CREATE DATABASE main; USE main; CREATE TABLE user ( user_id int, address_street varchar(50), address_city varchar(30), address_state varchar(2), address_zip int, user_type varchar(50), name_first varchar(50), name_last varchar(50), user_password varchar(20), user_email varchar(50), user_phone int ); USE main; CREATE TABLE donation ( donation_id int, product_name varchar(150), product_quantity int, product_quantity_measurement varchar(20), donator_user_id int, expiration varchar(30), comments varchar(200), donation_status varchar(50) DEFAULT 'open', pickup_user_id int );
